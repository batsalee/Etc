#include <string>
#include <vector>
#include <cmath>
#include <bitset>
#include <iostream>

using namespace std;

string solution(string bin1, string bin2) {
    string answer = "";
    
    /*
    2진수를 10진수로 변환 후 더하고 다시 2진수로 바꾸는 방식
    */
    int b1 = stoi(bin1);
    int b2 = stoi(bin2);
    int decimal;
    
    // bin1을 10진수로
    decimal = 0;
    for(int i=0; b1>=1; i++)
    {
        decimal += b1%10 * pow(2,i);
        b1 /= 10;
    }
    b1 = decimal;
    
    // bin2을 10진수로
    decimal = 0;
    for(int i=0; b2>=1; i++)
    {
        decimal += b2%10 * pow(2,i);
        b2 /= 10;
    }
    b2 = decimal;
    
    // 10진수로 바꾼 두 값을 더해주고
    int sum = b1+b2;
    
    // 10진수를 다시 2진수로
    if(sum != 0)
    {
        for(int i=0; sum>=1; i++)
        {
            answer = to_string(sum%2) + answer;
            sum /= 2;
        }
    }
    else if(sum == 0)
        answer = "0";
    
    return answer;
}
